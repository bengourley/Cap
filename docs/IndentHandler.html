Decrements the current indent.
If the indentation level is already at

	this.prevIndent = function () {
		if (this.indent - 1 < 0) {
			throw {
				name : 'IndentException',
				message : 'Indentation level already at zero'
			};
		} else {
			this.indent--;
		}
	};
